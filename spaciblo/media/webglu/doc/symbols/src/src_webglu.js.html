385             this.parseShaderVariables = function(str) {
386                 var tokens = str.split(/[\s\n;]+?/);

388                 for (var i = 0; i < tokens.length; i++) {
389                     if (tokens[i] == "attribute") {

402                             this.setModelViewUniform($W.constants.ModelViewUniform);

404                         }else if (name == $W.constants.ProjectionUniform) {
405                             this.setProjectionUniform($W.constants.ProjectionUniform);

633             /** Link this shader program to make it useable.
634              * Will [re]compile all attached shaders if necessary.
635              */
636             this.link = function() {
637                 console.group("linking program `" + this.name + "`");
638 
639 
640                 // Only delete the program if one already exists
641                 if (this.glProgram !== null) {
642                     console.log("already exists, deleting and relinking");
643                     $W.GL.deleteProgram(this.glProgram);
644                     this.attributes = [];
645                     this.glProgram = null;
646                 }
647 
648                 this.glProgram = $W.GL.createProgram();
649 
650                 // Attach all the shaders
651                 for (var i = 0; i < this.shaders.length; i++) {
652                     var shader = $W.shaders[this.shaders[i]];
653                     var glShader = shader.getGLShader();
654 
655                     // if the shader is still dirty after calling get,
656                     // which should have cleaned it, then the compile failed.
657                     if (shader.isDirty()) {
658                         console.log(this.shaders[i] + " failed to compile");
659                     }else {
660                         $W.GL.attachShader(this.glProgram, 
661                                 $W.shaders[this.shaders[i]].getGLShader());
662                     }	
663                 }
664 
665                 $W.GL.linkProgram(this.glProgram);
666 
667                 // Check for errors
668                 if (!$W.GL.getProgramParameter(this.glProgram, $W.GL.LINK_STATUS)) {
669                     console.group('Link error');
670                     console.error($W.GL.getProgramInfoLog(this.glProgram));
671                     dirty();
672                     console.groupEnd();
673                 }
674                     
675                 clean();
676                 this._setupAttributes();
677                 this._setupUniforms();
678 
679                 console.groupEnd();
680                 return isDirty;
681             }

1071             var cosHalfTheta = q1.w * q2.w + q1.x * q2.x + q1.y * q2.y + q1.z * q2.z;

2791  * at triangles formed by every face and averaging.
2792  * (c) 2009 Vladimir Vukicevic
2793  */
2794 $W.util.calculateNormals = function(vertices, faces) {
2795     var nvecs;
2796 
2797     if (vertices[0].length == 3) {
2798         nvecs = new Array(vertices.length);
2799 
2800         for (var i = 0; i < faces.length; i++) {
2801             var j0 = faces[i][0];
2802             var j1 = faces[i][1];
2803             var j2 = faces[i][2];
2804 
2805             var v1 = $V(vertices[j0]);
2806             var v2 = $V(vertices[j1]);
2807             var v3 = $V(vertices[j2]);
2808 
2809 
2810             var va = v2.subtract(v1);
2811             var vb = v3.subtract(v1);
2812 
2813             var n = va.cross(vb).toUnitVector();
2814 
2815             if (!nvecs[j0]) nvecs[j0] = [];
2816             if (!nvecs[j1]) nvecs[j1] = [];
2817             if (!nvecs[j2]) nvecs[j2] = [];
2818 
2819             nvecs[j0].push(n);
2820             nvecs[j1].push(n);
2821             nvecs[j2].push(n);
2822         }
2823 
2824     }else { // handle flattened arrays
2825         nvecs = new Array(vertices.length / 3)
2826 
2827         for (var i = 0; i < faces.length; i+=3) {
2828             var j0 = faces[i+0];
2829             var j1 = faces[i+1];
2830             var j2 = faces[i+2];
2831 
2832             var v1 = $V([vertices[j0], vertices[j0+1], vertices[j0+2]]);
2833             var v2 = $V([vertices[j1], vertices[j2+1], vertices[j2+2]]);
2834             var v3 = $V([vertices[j2], vertices[j2+1], vertices[j2+2]]);
2835 
2836 
2837             var va = v2.subtract(v1);
2838             var vb = v3.subtract(v1);
2839 
2840             var n = va.cross(vb).toUnitVector();
2841 
2842             if (!nvecs[j0]) nvecs[j0] = [];
2843             if (!nvecs[j1]) nvecs[j1] = [];
2844             if (!nvecs[j2]) nvecs[j2] = [];
2845 
2846             nvecs[j0].push(n);
2847             nvecs[j1].push(n);
2848             nvecs[j2].push(n);
2849         }
2850     }
2851 
2852     var normals = new Array(vertices.length);
2853 
2854     // now go through and average out everything
2855     for (var i = 0; i < nvecs.length - 1; i++) {
2856         var count = nvecs[i].length;
2857         var x = 0;
2858         var y = 0;
2859         var z = 0;
2860 
2861         for (var j = 0; j < count; j++) {
2862             x += nvecs[i][j].elements[0];
2863             y += nvecs[i][j].elements[1];
2864             z += nvecs[i][j].elements[2];
2865         }
2866 
2867         normals[i] = [x/count, y/count, z/count];
2868     }
2869 
2870     return normals;

 78         _initialize:function() {
 79             var BIND = addEventListener;
 80             var actions = $G.event._actions;
 81 
 82             BIND("keydown", function(ev) {
 83                 if (actions['+' + ev.keyCode] !== undefined) {
 84                     actions['+' + ev.keyCode]();
 85                 } }, false);
 86 
 87             BIND("keyup", function(ev) {
 88                 if (actions['-' + ev.keyCode] !== undefined) {
 89                     actions['-' + ev.keyCode]();
 90                 } }, false);
 91 
 92             var M = $G.mouse;
 93             actions['+mouse'] = [];
 94             actions['+mouse'][M.LB] = function(x,y){};
 95             actions['+mouse'][M.RB] = function(x,y){};
 96 
 97             actions['-mouse'] = [];
 98             actions['-mouse'][M.LB] = function(x,y){};
 99             actions['-mouse'][M.RB] = function(x,y){};
100 
101             actions['mousemove'] = function(x,y){};
102 
103             BIND("mousedown", function(ev) {
104                 console.dir(ev);
105                 actions['+mouse'][ev.which](
106                     ev.clientX - $G.offsetX, ev.clientY - $G.offsetY);
107             }, false);
108 
109             BIND("mouseup", function(ev) {
110                 actions['-mouse'][ev.which](
111                     ev.clientX -  $G.offsetX, ev.clientY -  $G.offsetY);
112             }, false);
113 
114             BIND("mousemove", function(ev) {

117             }, false);

122         bindOnElement:function(element, eventName, action) {

127         /* Add this key to the list of bound keys so we can only
128          * try to call functions for keys that have been bound.
129          *
130          * Associate the corresponding keycode with the passed function
131          * in the function array.
132          *
133          * '+keyname' or 'keyname' binds to the keydown event
134          * '-keyname' binds to keyup
135          * @param
136          */
137         bind:function(key, action) {
138             var M = $G.mouse
139             var actions = $G.event._actions;
140 
141             // Elements
142             if (arguments.length === 3) {
143                 arguments[0].addEventListener(arguments[1], arguments[2]);
144 
145             // Mousebuttons XXX Kludge?
146             }else if (key == "+m1" || key == "m1") {
147                 actions['+mouse'][M.LB] = action;
148 
149             }else if (key == "+m2" || key == "m2") {
150                 actions['+mouse'][M.RB] = action;
151 
152             }else if (key == "-m1") {
153                 actions['-mouse'][M.LB] = action;
154 
155             }else if (key == "-m2") {
156                 actions['-mouse'][M.RB] = action;
157 
158             }else if (key == "mousemove") {
159                 actions['mousemove'] = action;
160 
161             // Keys
162             }else if (key[0] == "+" || key[0] == "-") {
163 
164                 if (key[0] == "+") {
165                     key = $G.keys[key.slice(1)];
166                     actions['+' + key] = action;
167 
168                 }else if (key[0] == "-") {
169                     key = $G.keys[key.slice(1)];
170                     actions['-' + key] = action;
171                 }
172 
173             // Default to keydown
174             }else {
175                 key = $G.keys[key];
176                 actions['+' + key] = action;
177             }
178         },
179     },

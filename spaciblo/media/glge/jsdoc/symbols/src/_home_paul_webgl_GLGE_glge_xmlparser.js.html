 30 var XML={};
 31 (function(XML){
 32 XML.TextNode=function(value){
 33 	this.nodeValue=value;
 34 };
 35 XML.Element=function(name){
 36 	this.tagName=name;
 37 	this.children=[];
 38 	this.attributes={};
 39 };
 40 XML.Element.prototype.children=null;
 41 XML.Element.prototype.attributes=null;
 42 XML.Element.prototype.addChild=function(node){
 43 	node.parentNode=this;
 44 	if(this.children.length==0){
 45 		this.hasChildren=true;
 46 		this.firstChild=node;
 47 	}else{
 48 		this.children[this.children.length-1].nextSibling=node;
 49 		node.previousSibling=this.children[this.children.length-1];
 50 	}
 51 	this.children.push(node);
 52 };
 53 XML.Element.prototype.setAttributes=function(attributes){
 54 	this.attributes=attributes;
 55 };
 56 XML.Element.prototype.getAttribute=function(name){
 57 	return this.attributes[name];
 58 };
 59 XML.Element.prototype.hasAttribute=function(name){
 60 	if(this.attributes[name]) return true;
 61 		else return false;
 62 };
 63 //functioin returns array of elements
 64 XML.Element.prototype.getElementsByTagName=function(tagName){
 65 	var retArray=[];
 66 	for(var i=0;i<this.children.length;i++){
 67 		if(this.children[i].tagName==tagName || tagName=="*"){
 68 			if(this.children[i].tagName) retArray.push(this.children[i]);
 69 		}
 70 		//process the children
 71 		if(this.children[i].hasChildren) retArray=retArray.concat(this.children[i].getElementsByTagName(tagName));
 72 	}
 73 	return retArray;
 74 };
 75 XML.Document=function(xml){
 76 	this.currentElement=null;
 77 	this.rootElement=null;
 78 	this.eleStack=[];
 79 	this.idCache={};
 80 	this.xml=xml;
 81 	this.sPointer=0;
 82 	this.parseXML(xml);
 83 };
 84 XML.Document.prototype.parseXML=function(){
 85 	xml=this.xml;
 86 	textValue="";
 87 	cdata=false;
 88 	do{
 89 		if(xml[this.sPointer]=="<" && !cdata){
 90 			//add the text node
 91 			if(textValue!="" && this.currentElement) this.currentElement.addChild(new XML.TextNode(textValue));
 92 			textValue="";
 93 			if(xml[this.sPointer+1]=="!"){
 94 				//comment or cdata?
 95 				if(xml[this.sPointer+2]=="[" && xml[this.sPointer+3]=="C" && xml[this.sPointer+4]=="D" && xml[this.sPointer+5]=="A" && xml[this.sPointer+6]=="T" && xml[this.sPointer+7]=="A" && xml[this.sPointer+8]=="["){
 96 					cdata=true;
 97 					this.sPointer=this.sPointer+8;
 98 				}else{
 99 					//it's just a comment so forget it
100 					this.closeElement();
101 					this.sPointer++;
102 				}
103 			}else if(xml[this.sPointer+1]=="?"){
104 				this.closeElement();
105 				this.sPointer++;
106 			}else if(xml[this.sPointer+1]!="/"){
107 				//new element
108 				newElement=this.parseElement();
109 				if(xml[this.sPointer]==">"){
110 					this.eleStack.push(this.currentElement);
111 					this.currentElement=newElement;
112 				}else{
113 					//self closing
114 					this.currentElement.addChild(newElement);
115 					this.sPointer++;
116 				}
117 			}else{
118 				//close element
119 				parentElement=this.eleStack.pop();
120 				if(parentElement){
121 					parentElement.addChild(this.currentElement);
122 					this.currentElement=parentElement;
123 				}else{
124 					this.rootElement=this.currentElement;
125 				}
126 				this.closeElement();
127 			}
128 		}else{
129 			if(cdata && xml[this.sPointer]=="]" && xml[this.sPointer+1]=="]"  && xml[this.sPointer+2]==">"){
130 				//close cdata section!
131 				this.currentElement.addChild(new XML.TextNode(textValue));
132 				textValue="";
133 				this.sPointer=this.sPointer+2;
134 				cdata=false;
135 			}else{
136 				textValue=textValue+xml[this.sPointer];
137 				if(textValue.length==32768){
138 					//add the text node
139 					this.currentElement.addChild(new XML.TextNode(textValue));
140 					textValue="";
141 				}
142 			}
143 		}
144 		this.sPointer++;
145 	}while(this.sPointer<xml.length);
146 };
147 XML.Document.prototype.closeElement=function(){
148 	var xml=this.xml;
149 	do{
150 		this.sPointer++;
151 	}while(xml[this.sPointer]!=">" && this.sPointer<xml.length);
152 };
153 XML.Document.prototype.parseElement=function(){
154 	var xml=this.xml;
155 	var name="";
156 	this.sPointer++;
157 	do{
158 		if(xml[this.sPointer]==" "){
159 			var attributes=this.parseAttributes();
160 		}else{
161 			name=name+xml[this.sPointer];
162 			this.sPointer++;
163 		}
164 	}while(xml[this.sPointer]!="/" && xml[this.sPointer]!=">" && this.sPointer<xml.length);
165 	var element=new XML.Element(name);
166 	if(attributes){
167 		element.setAttributes(attributes);
168 		if(attributes.id){
169 			this.idCache[attributes.id]=element;
170 		}
171 	}
172 	return element;
173 };
174 XML.Document.prototype.parseAttributes=function(){
175 	var xml=this.xml;
176 	var attributes={};
177 	do{
178 		var name="";
179 		do{
180 			if(xml[this.sPointer]!=" ") name=name+xml[this.sPointer];
181 			this.sPointer++;
182 		}while(xml[this.sPointer]!="=" && this.sPointer<xml.length);
183 		var quote=xml[++this.sPointer];
184 		var value="";
185 		this.sPointer++;
186 		do{
187 			value=value+xml[this.sPointer];
188 			this.sPointer++;
189 		}while(xml[this.sPointer]!=quote && this.sPointer<xml.length);
190 		if(name!="") attributes[name]=value;
191 		this.sPointer++;
192 	}while(xml[this.sPointer]!="/" && xml[this.sPointer]!=">" && this.sPointer<xml.length);
193 	return attributes;
194 };
195 XML.Document.prototype.getElementsByTagName=function(tagName){
196 	return this.rootElement.getElementsByTagName(tagName);
197 };
198 XML.Document.prototype.getElementById=function(id){
199 	if(this.idCache[id]){
200 		return this.idCache[id];
201 	}else{
202 		return false;
203 	}
204 };
205 })(XML);
